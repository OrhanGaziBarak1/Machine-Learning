import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.preprocessing import StandardScaler

# Veri setini ykle
data = pd.read_excel("yeni_veri_eklenmis.xlsx")

# znitelikleri ve hedef deikeni ayr
X = data[['band5', 'band10', 'max_pow_time','red_hist_mean','green_hist_mean']]  # Seilen stunlar
y = data['rate']  # kt deikeni

# Veri setini standartlatr
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Standartlatrlm veriyi DataFrame'e dntr
X_scaled_df = pd.DataFrame(X_scaled, columns=X.columns)

# Her bir stun iin farkl grafik trleri ile grselletirme
for column in X_scaled_df.columns:
    plt.figure(figsize=(12, 6))
    
    # Bar plot
    plt.subplot(2, 2, 1)
    sns.barplot(x=X_scaled_df[column], y=y)
    plt.xlabel(column)
    plt.ylabel('rate')
    plt.title(f'{column} vs rate (Bar Plot)')

    # Scatter plot
    plt.subplot(2, 2, 2)
    sns.scatterplot(x=X_scaled_df[column], y=y)
    plt.xlabel(column)
    plt.ylabel('rate')
    plt.title(f'{column} vs rate (Scatter Plot)')

    # Violin plot
    plt.subplot(2, 2, 3)
    sns.violinplot(x=X_scaled_df[column], y=y)
    plt.xlabel(column)
    plt.ylabel('rate')
    plt.title(f'{column} vs rate (Violin Plot)')

    # Box plot
    plt.subplot(2, 2, 4)
    sns.boxplot(x=X_scaled_df[column], y=y)
    plt.xlabel(column)
    plt.ylabel('rate')
    plt.title(f'{column} vs rate (Box Plot)')

    plt.tight_layout()
    plt.show()

# znitelikler arasndaki ilikiyi gsteren heatmap
plt.figure(figsize=(10, 8))
corr = X_scaled_df.corr().round(2)
mask = np.triu(np.ones_like(corr,dtype=bool))
sns.heatmap(corr, annot=True, cmap='coolwarm', mask=mask, fmt=".0f")
plt.title('znitelikler Arasndaki Korelasyon Heatmap')
plt.show()

import pandas as pd
from sklearn.model_selection import KFold, cross_val_score, train_test_split
from sklearn.linear_model import LinearRegression, Ridge, Lasso, ElasticNet
from sklearn.metrics import r2_score, mean_squared_error, mean_absolute_error
from math import sqrt
import joblib


data = pd.read_excel('standardize_edilmis_veri.xlsx')



# Bamsz deikenler ve hedef deikeni ayr
X = data.drop(columns=["rate"])
y = data["rate"]

# Kullanlacak modelleri tanmla
models = {
    "Linear Regression": LinearRegression(),
    "Ridge Regression": Ridge(),
    "Lasso Regression": Lasso(),
}

# Her model iin performans deerlendir ve en iyisini se
best_model = None
best_score = float('-inf')
for name, model in models.items():
    cv_scores = cross_val_score(model, X, y, cv=10, scoring='r2')
    mean_r2 = cv_scores.mean()
    print(f"{name} cross-validation mean R^2 score:", mean_r2)
    if mean_r2 > best_score:
        best_score = mean_r2
        best_model = model

# En iyi modeli eit
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=47)
best_model.fit(X_train, y_train)

# Eitim setinde performans deerlendir
train_predictions = best_model.predict(X_train)
train_r2 = r2_score(y_train, train_predictions)
train_mse = mean_squared_error(y_train, train_predictions)
train_mae = mean_absolute_error(y_train, train_predictions)
train_rmse = sqrt(train_mse)
print("\nEitim Seti Performans:")
print("R^2 deeri:", train_r2)
print("MSE (Ortalama Kare Hata):", train_mse)
print("MAE (Ortalama Mutlak Hata):", train_mae)
print("RMSE (Kk Ortalama Kare Hata):", train_rmse)

# Test setinde performans deerlendir
test_predictions = best_model.predict(X_test)
test_r2 = r2_score(y_test, test_predictions)
test_mse = mean_squared_error(y_test, test_predictions)
test_mae = mean_absolute_error(y_test, test_predictions)
test_rmse = sqrt(test_mse)
print("\nTest Seti Performans:")
print("R^2 deeri:", test_r2)
print("MSE (Ortalama Kare Hata):", test_mse)
print("MAE (Ortalama Mutlak Hata):", test_mae)
print("RMSE (Kk Ortalama Kare Hata):", test_rmse)

# En iyi modeli kaydet
joblib.dump(best_model, 'best_model_isedataset_regression.pkl')
